#对恒星炮
planet_event = {
	id = sfdoll_planet_killer_event.1
	hide_window = yes
	is_triggered_only = yes
	immediate = {
		#威慑值和得分
		add_threat = { who = from.owner amount = 10 }
		from = {
			owner = {
				#save_event_target_as = crisis_country
				add_victory_score = {
					source = destroyed_stars_score
					score = 10
				}
			}
		}
		#处理殖民地
		if = {
			limit = { is_colony = yes }
			every_owned_pop = {
				kill_pop = yes
			}
			destroy_colony = yes
		}

		#炸的是恒星
		if = {
			limit =  { is_star = yes }
			from = {
				owner = {
					#初次炸恒星
					if = {
						limit = {
							NOT = { has_country_flag = cracked_first_star }
						}
						set_country_flag = cracked_first_star
					}
				}
			}
			#变成黑洞
			# change_pc = pc_black_hole
			# clear_deposits = yes
			# reset_planet = yes
			solar_system = { save_event_target_as = destroyed_system }
			solar_system = {
				destroy_star_system = yes
			}
		}
		else_if = {
			#环世界
			limit  = {
				is_ringworld = yes
			}
			change_pc = pc_shattered_ring_habitable
			clear_deposits = yes
			
		}
		else = {
			change_pc = pc_shattered
			clear_deposits = yes
		}
	}
}

sfdoll_habitat_01 ={
	entity = "construction_platform_entity"
	construction_entity = "construction_platform_entity"
	show_galactic_map_icon = yes
	place_entity_on_planet_plane = no
	build_outside_gravity_well = yes
	build_time = 360
	construction_blocks_and_blocked_by = multi_stage_type
	potential = {
		owner = {
			has_authority = auth_gf_sfdoll
		}
	}
	show_prereqs = yes
	prerequisites = {
		tech_sfdoll_habitat_01
	}
	possible = {
		exists = starbase
		custom_tooltip = {
			fail_text = "requires_inside_border"
			is_inside_border = from
		}
		custom_tooltip = {
			fail_text = "requires_surveyed_system"
			NOT = {
				any_system_planet = {
					is_surveyed = {
						who = prev.from
						status = no
					}
				}
			}
		}

		#不能有环世界
		NOT = {
			OR = {
				has_star_flag = ring_world_built
				any_system_planet = {
					is_planet_class = pc_ringworld_habitable
					is_planet_class = pc_ringworld_habitable_damaged
					is_planet_class = pc_ringworld_tech
					is_planet_class = pc_ringworld_tech_damaged
					is_planet_class = pc_ringworld_seam
					is_planet_class = pc_ringworld_seam_damaged
				}
			}
		}
	}

	resources = {
		category = megastructures_habitat
		cost = {
			alloys = 5000
			nanites = 500
			unity = 150
		}
	}
	#Solar System scope, FROM = builder empire, FROMFROM = the megastructure
	on_build_complete = {
		from = {
			#set_country_flag = has_built_or_repaired_megastructure
			create_message = {
				type = MEGASTRUCTURE_BUILT
				localization = sfdoll_habitat_01_complete
				days = 1800
				target= prev
			}
		}
	}

	ai_weight = {
		weight = 0
	}
}

sfdoll_habitat_02 ={
	entity = ""
	construction_entity = "construction_platform_entity"
	show_galactic_map_icon = yes
	portrait = "GFX_sfdoll_habitat_01"
	place_entity_on_planet_plane = no
	build_time = 1800
	construction_blocks_and_blocked_by = multi_stage_type
	potential = {
		owner = {
			has_authority = auth_gf_sfdoll
		}
	}
	upgrade_from = {
		sfdoll_habitat_01
	}

	resources = {
		category = megastructures_habitat
		cost = {
			alloys = 20000
			nanites = 2000
			unity = 450
		}
	}
	#Solar System scope, FROM = builder empire, FROMFROM = the megastructure
	on_build_complete = {
		spawn_planet = {
			class = pc_sfdoll_habitis
			location = fromfrom
			size = 10
			has_ring = yes
			init_effect = {
				set_name = "HABITAT_PLANET_NAME"
				set_surveyed = {
					surveyed = yes
					surveyor = FROM
				}
				set_all_comms_surveyed = yes
				clear_blockers = yes
				start_colony = {
					owner = root
				}
				set_planet_entity = {
					entity = GF_nyto_habita_entity
				}
				save_event_target_as = this_planet_habiti
				set_planet_flag = megastructure
				set_planet_flag = habitat
			}
		}
		from = {
			set_country_flag = has_built_or_repaired_megastructure
			create_message = {
				type = MEGASTRUCTURE_BUILT
				localization = sfdoll_habitat_02_complete
				days = 30
				target = prev
			}
		}
		fromfrom = {
			remove_megastructure = this
		}
	}

	ai_weight = {
		weight = 0
	}
}
